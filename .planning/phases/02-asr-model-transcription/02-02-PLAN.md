**Step 1: Define SentenceManager class**

```typescript
const SENTENCE_END_PUNCTUATION = /[。！？\.!\?]$/;

class SentenceManager {
  private partialText = "";
  private plugin: LinkTagIntelligencePlugin;

  constructor(plugin: LinkTagIntelligencePlugin) {
    this.plugin = plugin;
  }

  /** Called on every Worker result — accumulates partial text. */
  addPartialText(text: string): void {
    this.partialText += text;
  }

  /** Called when isEndpoint=true — finalizes the current sentence. */
  finalizeSentence(text: string): string {
    // Use the accumulated text if no explicit final text is provided
    const sentence = text || this.partialText;
    this.partialText = "";

    // D-09: Add trailing punctuation if not already present
    const trimmed = sentence.trim();
    if (!trimmed) return "";

    if (SENTENCE_END_PUNCTUATION.test(trimmed)) {
      return trimmed;
    }
    // Default: append Chinese period for zh, English period for en
    const lang = this.plugin.settings.speechLanguage;
    return trimmed + (lang === "zh" ? "。" : ".");
  }

  /** Get current accumulated partial text (for debugging). */
  getPartialText(): string {
    return this.partialText;
  }

  /** Reset on recording stop. */
  reset(): void {
    this.partialText = "";
  }
}
```

**Step 2: Extend toggleSpeechRecording() in main.ts**

Replace the placeholder `onAsrResult` callback in `toggleSpeechRecording()` with SentenceManager integration:

```typescript
async toggleSpeechRecording(): Promise<void> {
  if (!this.speechRecorder.canToggle()) {
    if (this.speechRecorder.getSnapshot().phase === "error") {
      this.speechRecorder.acknowledgeError();
      this.refreshAllViews();
    }
    return;
  }

  // Create SentenceManager on first use
  if (!this._sentenceManager) {
    this._sentenceManager = new SentenceManager(this);
  }

  // Set ASR result handler
  this.speechRecorder.onAsrResult = (text, isEndpoint) => {
    if (isEndpoint) {
      // D-06: Sentence boundary — finalize and insert at cursor (D-07)
      const finalSentence = this._sentenceManager!.finalizeSentence(text);
      if (finalSentence) {
        this.insertSpeechText(finalSentence);
      }
    } else {
      // Partial result — accumulate only (no insertion per deferred decision)
      this._sentenceManager!.addPartialText(text);
    }
  };

  const errorKey = await this.speechRecorder.toggle((key, vars) => this.t(key as Parameters<typeof tr>[1], vars));

  if (errorKey) {
    new Notice(this.t(errorKey as Parameters<typeof tr>[1]));
    this._sentenceManager?.reset();
    this.refreshAllViews();
    return;
  }

  // On stop: flush any remaining partial text
  if (!this.speechRecorder.isActive && this._sentenceManager) {
    const remaining = this._sentenceManager.getPartialText();
    if (remaining.trim()) {
      const final = this._sentenceManager.finalizeSentence(remaining);
      if (final) this.insertSpeechText(final);
    }
    this._sentenceManager.reset();
  }

  this.refreshAllViews();
}
```

**Step 3: Add insertSpeechText method (D-07, D-08)**

Add a private method to LinkTagIntelligencePlugin:

```typescript
private insertSpeechText(text: string): void {
  if (!text) return;

  const editorView = this.getContextEditorView();
  if (!editorView?.editor) {
    debugLog(this.app, "speech.insert-no-editor", { text });
    return;
  }

  const editor = editorView.editor;

  // D-08: Save cursor position before insert (for concurrent typing protection)
  const cursor = editor.getCursor();

  // D-07: Insert sentence at current cursor with trailing space
  editor.replaceSelection(text + " ");

  // D-08: Cursor is already positioned after inserted text by replaceSelection.
  // If the cursor moved due to concurrent input between save and replace,
  // the user's manual cursor position takes precedence — no forced restoration.
  debugLog(this.app, "speech.text-inserted", {
    textLen: text.length,
    cursorLine: cursor.line,
    cursorCh: cursor.ch,
  });
}
```

```typescript
private buildSpeechButtonSnapshot(
  key: ToolbarActionId,
  snapshot: RecorderSnapshot
): ToolbarButtonSnapshot {
  // Determine tooltip based on phase AND ASR status
  const asrLoading = snapshot.phase === "initializing" || 
    (snapshot.phase === "recording" && !snapshot.asrReady);
  
  let tooltipKey: string;
  let label = this.plugin.t("speechRecord");

  if (snapshot.phase === "error") {
    tooltipKey = snapshot.errorKey === "speechAsrInitFailed" 
      ? "speechAsrError" 
      : "speechRecordTooltipError";
  } else if (asrLoading) {
    tooltipKey = "speechAsrLoading";
  } else if (snapshot.phase === "processing") {
    tooltipKey = "speechRecordTooltipProcessing";
  } else if (snapshot.phase === "recording") {
    // D-12: Show countdown when <= 10s remaining
    const remaining = this.plugin.getAutoStopSecondsRemaining();
    if (remaining > 0 && remaining <= 10) {
      label = `${remaining}s`;  // Button text becomes countdown
      tooltipKey = "speechAsrAutoStopCountdown";
    } else {
      tooltipKey = "speechRecordTooltipRecording";
    }
  } else if (snapshot.asrReady) {
    tooltipKey = "speechAsrReady";  // idle + model ready
  } else {
    tooltipKey = "speechRecordTooltipIdle";
  }

  const isError = snapshot.phase === "error";

  return {
    key,
    label,
    disabled: false,
    title: this.plugin.t(tooltipKey as any, 
      tooltipKey === "speechAsrAutoStopCountdown" 
        ? { seconds: this.plugin.getAutoStopSecondsRemaining() } 
        : undefined),
    state: snapshot.phase,
    audioLevel: snapshot.phase === "recording" ? snapshot.audioLevel : undefined,
    dbValue: snapshot.phase === "recording" ? snapshot.dbValue : undefined,
  };
}
```
